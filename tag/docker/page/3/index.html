docker更换存储位置
2024-11-11 # docker

docker info | grep "Docker Root Dir"
默认在/var/lib/docker
systemctl stop docker
mv /var/lib/docker /data/docker
ln -sf /data/docker /var/lib/docker
重启docker
systemctl start docker
