Posted on 2024-11-24，1 min read

安装

mkdir -p /data/leanote/db
mkdir -p /data/leanote/conf
mkdir -p /data/leanote/files
mkdir -p /data/leanote/public/upload

docker run -d --name leanote \
    -v /data/leanote/db:/data/db \
    -v /data/leanote/conf/:/data/leanote/conf \
    -v /data/leanote/files:/data/leanote/files \
    -v /data/leanote/public/upload:/data/leanote/public/upload \
    -p 9000:9000 --restart unless-stopped \
    axboy/leanote
