samba匿名共享

2024-11-07 1 min read # linux

安装samba
修改/etc/samba/smb.conf

[global]
       workgroup = SAMBA
        security = user
        map to guest = bad user
        guest ok = yes
[disk1]
	path = /disk1
	browseable =yes
	public=yes
	writable=yes
	create mask=0777
	directory mask=0777
